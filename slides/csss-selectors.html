CSS Selectors

    
/* Element of type span */
span {}

/* Element span with class 'myclass' */
span.myclass {}

/* Element span with attribute equals */
span.myclass[foo='bar'] {}

/* Element span, the n-th child of parent */
span:nth-child(6) {}

/* Element span, pseudo-class hover (e.g. mouseover) */
span.myclass:hover {}
